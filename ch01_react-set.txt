# touch 설치
- 터미널에서 touch 명령어를 사용해서 새로운 파일을 만들거나 기존 파일을 변경할 수 있습니다
$ npm install -g touch-cli
$ touch --version
// $ 터미널 명령어

# 폴더(디렉토리) 생성
$ mkdir 폴더명 

#touch 파일생성
$ touch 파일명

#  폴더(디렉토리), 파일 삭제
$ rm -r -force 삭제폴더 // window
$ rmdir 폴더명
$  del 파일명
$ rmdis /s 폴더명

$ rm -rf 삭제폴더       //mac
